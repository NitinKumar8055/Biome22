void loop()
{
   crying();

 if(digitalRead(2)==HIGH&&digitalRead(3)==LOW){
  digitalWrite(11,HIGH);
  digitalWrite(12,LOW);
  Serial.println("AIR pump is on");
  }
  
  else if(digitalRead(3)==HIGH&&digitalRead(2)==LOW){
   digitalWrite(11,HIGH);
   digitalWrite(12,LOW);
    Serial.println("WATER pump is on");
    }
  else if(digitalRead(2)==LOW&&digitalRead(3)==LOW)
    {
     digitalWrite(11,HIGH);
     digitalWrite(12,HIGH);
     Serial.println("BOTH pump are on");
}
  else{
    
   digitalWrite(11,LOW);
   digitalWrite(12,LOW);
   Serial.println("BOTH pump are off");
  }
  
  oled();
    
delay(1000);


  
}
